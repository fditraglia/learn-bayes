This example is taken from the STAN manual article on seemingly unrelated regressions (SUR). Strictly speaking, this isn’t really an SUR model at all since the regressors are the same in each equation. Instead it’s a multivariate regression model.

First is our STAN model:

library(cmdstanr)
# this is in ~/learn-bayes/STAN-examples
SUR_model <- cmdstan_model("SUR.stan")
SUR_model$print()
data {
  int<lower=1> K;
  int<lower=1> J;
  int<lower=0> N;
  array[N] vector[J] x;
  array[N] vector[K] y;
}
parameters {
  matrix[K, J] beta;
  cholesky_factor_corr[K] L_Omega;
  vector<lower=0>[K] L_sigma;
}
model {
  array[N] vector[K] mu;
  matrix[K, K] L_Sigma;

  for (n in 1:N) {
    mu[n] = beta * x[n];

  }

  L_Sigma = diag_pre_multiply(L_sigma, L_Omega);

  to_vector(beta) ~ normal(0, 5);
  L_Omega ~ lkj_corr_cholesky(4);
  L_sigma ~ cauchy(0, 2.5);

  y ~ multi_normal_cholesky(mu, L_Sigma);
}

Next we generate some data:

library(mvtnorm)
set.seed(1234)

K <- 2 # number of outcome variables
J <- 5 # number of predictor variables
N <- 200 # sample size

x <- matrix(rnorm(n = N * J), nrow = N)
beta <- matrix(rnorm(n = J * K), nrow = K)

Sigma <- matrix(c(1, 0.5, 
                  0.5, 1), byrow = TRUE, nrow = 2)

errors <- rmvnorm(n = N, mean = rep(0, K), sigma = Sigma)
y <- x %*% t(beta) + errors

Next we fit the model:

dat <- list(N = N, K = K, J = J, y = y, x = x) 

SUR_model_fit <- SUR_model$sample(
  data = dat,
  seed = 123,
  chains = 4,
  parallel_chains = 4,
  refresh = 500
)
